% =================== MCQ 1 ===================
\begin{questiontitle}[MCQ 1]
\textbf{The main reason complex numbers were developed was to:}
\end{questiontitle}

\begin{partbox}[Options]
\begin{enumerate}[label=\Alph*.]
    \item Extend the real number system for unsolvable equations
    \item Create a new type of geometry
    \item Replace fractions and decimals
    \item Simplify basic arithmetic operations
\end{enumerate}
\end{partbox}

\begin{answerstyle}
\textbf{Answer: A} \\
Complex numbers were specifically developed to extend the real number system so that equations like \( x^2 + 1 = 0 \) would have solutions. These equations had no solutions within the realm of real numbers.
\end{answerstyle}

\begin{conceptbox}
\textbf{Historical Context:} Complex numbers emerged in the 16th century when mathematicians encountered equations like \( x^2 + 1 = 0 \). The introduction of \( i = \sqrt{-1} \) allowed such "impossible" equations to have solutions: \( x = \pm i \).
\end{conceptbox}

% =================== MCQ 2 ===================
\begin{questiontitle}[MCQ 2]
\textbf{Which equation type motivated the creation of complex numbers?}
\end{questiontitle}

\begin{partbox}[Options]
\begin{enumerate}[label=\Alph*.]
    \item Linear equations like \( 2x + 3 = 7 \)
    \item Quadratic equations like \( x^2 + 1 = 0 \)
    \item Simple fraction equations
    \item Basic algebraic expressions
\end{enumerate}
\end{partbox}

\begin{answerstyle}
\textbf{Answer: B} \\
Quadratic equations with negative discriminants, such as \( x^2 + 1 = 0 \), have no real solutions. Complex numbers were developed to provide solutions to these types of equations: \( x = \pm i \).
\end{answerstyle}

\begin{conceptbox}
\textbf{Key Insight:} When the discriminant \( b^2 - 4ac < 0 \) in the quadratic formula, we get \( \sqrt{\text{negative number}} \). Complex numbers make this possible through \( i = \sqrt{-1} \), giving us solutions like \( x = \frac{-b \pm i\sqrt{4ac - b^2}}{2a} \).
\end{conceptbox}

% =================== MCQ 3 ===================
\begin{questiontitle}[MCQ 3]
\textbf{The square root of a negative number in the realm of real numbers:}
\end{questiontitle}

\begin{partbox}[Options]
\begin{enumerate}[label=\Alph*.]
    \item Always equals zero
    \item Does not exist
    \item Equals the negative of the number
    \item Can be calculated using standard methods
\end{enumerate}
\end{partbox}

\begin{answerstyle}
\textbf{Answer: B} \\
In the real number system, there is no real number whose square gives a negative result. Since \( x^2 \geq 0 \) for all real \( x \), equations like \( x^2 = -1 \) have no real solutions.
\end{answerstyle}

\begin{conceptbox}
\textbf{Fundamental Limitation:} For any real number \( x \), \( x^2 \geq 0 \). This means \( \sqrt{-a} \) where \( a > 0 \) cannot exist in real numbers, creating the need for complex numbers.
\end{conceptbox}

% =================== MCQ 5 ===================
\begin{questiontitle}[MCQ 5]
\textbf{What happens when we try to find \( \sqrt{-9} \) using only real numbers?}
\end{questiontitle}

\begin{partbox}[Options]
\begin{enumerate}[label=\Alph*.]
    \item We get \( -3 \)
    \item We get \( 3 \)
    \item It is undefined/does not exist
    \item We get \( 9 \)
\end{enumerate}
\end{partbox}

\begin{answerstyle}
\textbf{Answer: C} \\
\( \sqrt{-9} \) is undefined in real numbers because there is no real number that when squared gives \( -9 \). Note that \( (-3)^2 = 9 \), not \( -9 \).
\end{answerstyle}

\begin{conceptbox}
\textbf{Common Misconception:} Students often think \( \sqrt{-9} = -3 \), but \( (-3)^2 = 9 \), not \( -9 \). This impossibility in real numbers led to the development of imaginary numbers where \( \sqrt{-9} = 3i \).
\end{conceptbox}

\textbf{In the complex number \( z = a + ib \), what do \( a \) and \( b \) represent?}
\end{questiontitle}

\begin{partbox}[Options]
\begin{enumerate}[label=\Alph*.]
    \item Both are imaginary numbers
    \item Both are real numbers
    \item \( a \) is real, \( b \) is imaginary
    \item \( a \) is imaginary, \( b \) is real
\end{enumerate}
\end{partbox}

\begin{answerstyle}
\textbf{Answer: B} \\
In the standard form \( z = a + ib \), both \( a \) and \( b \) are real numbers. Here \( a \) is called the real part and \( b \) is the coefficient of the imaginary unit \( i \).
\end{answerstyle}

\begin{conceptbox}
\textbf{Standard Form:} Every complex number can be written as \( z = a + ib \) where \( a, b \in \mathbb{R} \). The term \( a \) is the real part, \( ib \) is the imaginary part, but \( b \) itself is real.
\end{conceptbox}

    \item \( a = 5, b = 3 \)

\begin{answerstyle}
\textbf{Answer: B} \\

\end{answerstyle}

\begin{conceptbox}
\textbf{Sign Convention:} When a complex number is written as \( z = a - bi \), remember that \( b \) is negative. Always rewrite in the form \( a + ib \) to identify components correctly.
\end{conceptbox}
